def stem_token(word):

# ============ BM25 IMPLEMENTATION ============
class BM25:
    """BM25 ranking algorithm with stemming and synonym support."""

    def __init__(self, k1=1.5, b=0.75):
        self.k1 = k1
        self.b = b
        self.corpus = []
        self.doc_lengths = []
        self.avgdl = 0
        self.idf = {}
        self.doc_freqs = defaultdict(int)
        self.N = 0

    def tokenize(self, text):
        """Lowercase, split, remove punctuation, filter short words, stem."""
        text = re.sub(r'[^\w\s]', ' ', str(text).lower())
        tokens = [w for w in text.split() if len(w) > 2]
        return [stem_token(t) for t in tokens]

    def fit(self, documents):
        """Build BM25 index from documents."""
        self.corpus = [self.tokenize(doc) for doc in documents]
        self.N = len(self.corpus)
        if self.N == 0:
            return
        self.doc_lengths = [len(doc) for doc in self.corpus]
        self.avgdl = sum(self.doc_lengths) / self.N

        for doc in self.corpus:
            seen = set()
            for word in doc:
                if word not in seen:
                    self.doc_freqs[word] += 1
                    seen.add(word)

        for word, freq in self.doc_freqs.items():
            self.idf[word] = log((self.N - freq + 0.5) / (freq + 0.5) + 1)

    def score(self, query):
        """Score all documents against query, expanding synonyms."""
        raw_tokens = self.tokenize(query)
        query_tokens = [stem_token(t) for t in expand_synonyms(raw_tokens)]
        scores = []

        for idx, doc in enumerate(self.corpus):
            score = 0
            doc_len = self.doc_lengths[idx]
            term_freqs = defaultdict(int)
            for word in doc:
                term_freqs[word] += 1

            for token in query_tokens:
                if token in self.idf:
                    tf = term_freqs[token]
                    idf = self.idf[token]
                    numerator = tf * (self.k1 + 1)
                    denominator = tf + self.k1 * (1 - self.b + self.b * doc_len / self.avgdl)
                    score += idf * numerator / denominator

            scores.append((idx, score))

        return sorted(scores, key=lambda x: x[1], reverse=True)

def _load_csv(filepath):
    """Load CSV and return list of dicts."""
    with open(filepath, 'r', encoding='utf-8') as f:

def _search_csv(filepath, search_cols, output_cols, query, max_results):
    """Core search function using BM25 with caching."""
    cache_key = (str(filepath), query, max_results)
    if cache_key in _search_cache:
        return _search_cache[cache_key]

    if not filepath.exists():
        return []

    data = _load_csv(filepath)
    documents = [" ".join(str(row.get(col, "")) for col in search_cols) for row in data]

    bm25 = BM25()
    bm25.fit(documents)
    ranked = bm25.score(query)

    results = []
    for idx, score in ranked[:max_results]:
        if score > 0:
            row = data[idx]
            results.append({col: row.get(col, "") for col in output_cols if col in row})

    _search_cache[cache_key] = results
    return results

Domains: style, color, chart, landing, product, ux, typography, google-fonts, icons, react, web, anti-patterns
Stacks: dynamically discovered from data/stacks/*.csv
Intensity: wild (8-10), bold (5-7), refined (1-4), or a number 1-10

import io
from core import CSV_CONFIG, get_available_stacks, MAX_RESULTS, search, search_stack

# Force UTF-8 for stdout/stderr

    sys.stdout = io.TextIOWrapper(sys.stdout.buffer, encoding='utf-8')
if sys.stderr.encoding and sys.stderr.encoding.lower() != 'utf-8':
    sys.stderr = io.TextIOWrapper(sys.stderr.buffer, encoding='utf-8')


def parse_intensity(value):
    """Parse intensity from name or number."""
    if value is None:
        return None
    name_map = {"wild": 9, "bold": 6, "refined": 3}
    if value.lower() in name_map:
        return name_map[value.lower()]
    try:
        n = int(value)
        return max(1, min(10, n))
    except ValueError:
        return None


def format_output(result):
    """Format results for Claude consumption (token-optimized)."""
    if "error" in result:
        return f"Error: {result['error']}"

    output = []
    if result.get("stack"):
        output.append(f"## AI-HOSS Stack Guidelines")
        output.append(f"**Stack:** {result['stack']} | **Query:** {result['query']}")
    else:
        output.append(f"## AI-HOSS Search Results")
        output.append(f"**Domain:** {result['domain']} | **Query:** {result['query']}")
    output.append(f"**Source:** {result['file']} | **Found:** {result['count']} results\n")

    for i, row in enumerate(result['results'], 1):
        output.append(f"### Result {i}")
        for key, value in row.items():
            value_str = str(value)
            if len(value_str) > 300:
                value_str = value_str[:300] + "..."
            output.append(f"- **{key}:** {value_str}")
        output.append("")

    return "\n".join(output)

                if current_line != prefix:
                    lines.append(current_line)

        if current_line != prefix:

        return lines

# ============ PERSISTENCE ============
def persist_design_system(design_system, page=None, output_dir=None, page_query=None):
    """Persist design system using Master + Overrides pattern."""
    base_dir = Path(output_dir) if output_dir else Path.cwd()
    project_name = design_system.get("project_name", "default")
    project_slug = project_name.lower().replace(' ', '-')

    ds_dir = base_dir / "design-system" / project_slug
    pages_dir = ds_dir / "pages"
    ds_dir.mkdir(parents=True, exist_ok=True)
    pages_dir.mkdir(parents=True, exist_ok=True)

    # Write MASTER.md
    master_content = format_markdown(design_system)
    master_file = ds_dir / "MASTER.md"
    with open(master_file, 'w', encoding='utf-8') as f:
        f.write(f"<!-- AI-HOSS Master | Generated: {datetime.now().isoformat()} -->\n\n")
        f.write("> **LOGIC:** Check `pages/[page-name].md` first. If exists, it overrides this file.\n\n")
        f.write(master_content)

    created = [str(master_file)]

    # Write page override if specified
    if page:
        page_file = pages_dir / f"{page.lower().replace(' ', '-')}.md"
        with open(page_file, 'w', encoding='utf-8') as f:
            f.write(f"# {page.title()} Page Overrides\n\n")
            f.write(f"> Overrides MASTER.md for the {page} page.\n\n")
            f.write("## Overrides\n\n")
            f.write("_Add page-specific design overrides here._\n")
        created.append(str(page_file))

    return {"status": "success", "files": created}

## Design Guidelines

### Typography
- Choose fonts that are beautiful, unique, and characterful
- NEVER default to Inter, Roboto, Arial, or system fonts

- Use the typography search to find context-appropriate pairings:

- Commit to a cohesive palette with personality
- Dominant colors with sharp accents > timid, evenly-distributed palettes
- Use CSS variables for all color tokens (16-token system)
- Search for context-matched palettes:
  `python3 search.py "<product type>" --domain color`

### Motion & Animation
- Focus on high-impact moments: page load reveals, scroll triggers, hover states
- Match animation intensity to the dial:
  - REFINED: 150-250ms, ease-out, subtle transforms
  - BOLD: 200-400ms, custom easings, orchestrated sequences
  - WILD: 300-600ms+, spring physics, dramatic reveals, scroll-driven
- Prioritize CSS-only for HTML, Framer Motion for React, native APIs for mobile
- ALWAYS respect `prefers-reduced-motion`

### Spatial Composition
- REFINED: Clean grids, generous whitespace, clear hierarchy
- BOLD: Asymmetry, varied spacing, visual tension points
- WILD: Overlap, diagonal flow, grid-breaking, unexpected density

### Atmosphere & Texture
- Create depth and atmosphere, never flat solid colors
- Gradient meshes, noise textures, geometric patterns, layered transparencies
- Match to the aesthetic: grain for editorial, glass for modern, paper for organic

## Priority Rules

When guidelines conflict, follow this priority:

1. **Accessibility** (CRITICAL): Contrast 4.5:1, keyboard nav, aria-labels, alt text
2. **Touch & Interaction** (CRITICAL): 44x44pt targets, 8px+ spacing, loading feedback
3. **Performance** (HIGH): WebP/AVIF, lazy loading, CLS < 0.1, compositor animations
4. **Style Selection** (HIGH): Match product type, maintain consistency
5. **Layout & Responsive** (HIGH): Mobile-first, systematic breakpoints (375, 768, 1024, 1440)
6. **Typography & Color** (MEDIUM): 16px base, 1.5 line-height, semantic tokens
7. **Animation** (MEDIUM): 150-300ms duration, meaningful motion, spatial continuity
8. **Forms & Feedback** (MEDIUM): Visible labels, inline errors, helper text
9. **Navigation** (HIGH): Predictable back, bottom nav ≤5 items, deep linking
10. **Charts & Data** (LOW): Legends, tooltips, accessible colors

- **Inter/Roboto/Arial** as default fonts without intentional choice

- **Generic hero illustrations** — use real photography, 3D, or typography as hero
- **8px rounded corners on everything** — vary radii intentionally
- **Fade-in-up on scroll for everything** — choose context-specific animations
- **Blue CTA buttons** without palette context
- **Shadow-sm on every card** — vary elevation intentionally
- **Centered-everything layouts** — use asymmetry, alignment variety
- **System font stacks** without override — always specify fonts
